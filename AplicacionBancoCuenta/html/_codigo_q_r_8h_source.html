556 public: void generarPDFQR(const CodigoQR::QrCode& qr, const std::string& nombreArchivo) {
557 if (qr.getSize() == 0) {
558 std::cout << "Error: Primero debes generar el código QR" << std::endl;
559 return;
560 }
561
562 std::cout << "\nGenerando PDF..." << std::endl;
563
564 // Crear archivo PDF
565 std::ofstream pdf(nombreArchivo, std::ios::binary);
566 if (!pdf.is_open()) {
567 std::cout << "Error: No se pudo crear el archivo PDF." << std::endl;
568 return;
569 }
570
571 // Constantes para el PDF - convertir mm a puntos (1/72 pulgadas)
572 const auto MM_TO_POINTS = [](double mm) { return mm * 2.83465; };
573
574 // Tamaño de cada módulo QR en puntos
575 const double MODULO_SIZE = MM_TO_POINTS(8.0); // 8mm por módulo para mejor escaneo
576 const double MARGEN = MODULO_SIZE * 4; // Margen mínimo recomendado (4 módulos)
577 const double QR_SIZE = MODULO_SIZE * qr.getSize(); // Tamaño total del QR
578 const double PAGE_WIDTH = QR_SIZE + 2 * MARGEN;
579 const double PAGE_HEIGHT = PAGE_WIDTH; // Página cuadrada para mejor presentación
580 const double MARGEN_EXTRA = MARGEN * 0.5; // Margen adicional para mejor contraste
581
582 // Lista de objetos PDF
583 std::vector<size_t> objetos;
584 std::string xref;
585
586 // Escribir cabecera PDF
587 pdf << "%PDF-1.4\n%\xE2\xE3\xCF\xD3\n";
588
589 // Catálogo (objeto 1)
590 objetos.push_back(pdf.tellp());
591 pdf << "1 0 obj\n<</Type /Catalog /Pages 2 0 R>>\nendobj\n";
592
593 // Pages (objeto 2)
594 objetos.push_back(pdf.tellp());
595 pdf << "2 0 obj\n<</Type /Pages /Kids [3 0 R] /Count 1>>\nendobj\n";
596
597 // Page (objeto 3)
598 objetos.push_back(pdf.tellp());
599 pdf << "3 0 obj\n"
600 << "<</Type /Page /Parent 2 0 R "
601 << "/MediaBox [0 0 " << PAGE_WIDTH << " " << PAGE_HEIGHT << "] "
602 << "/Contents 4 0 R>>\nendobj\n";
603
604 // Contenido (objeto 4) - Aquí va el QR
605 std::stringstream contenido;
606 contenido << "q\n"; // Guardar estado gráfico
607
608 // Establecer color blanco y dibujar fondo
609 contenido << "1 1 1 rg\n"; // RGB blanco
610 contenido << "0 0 " << PAGE_WIDTH << " " << PAGE_HEIGHT << " re f\n";
611
612 // Establecer color negro sólido para el QR
613 contenido << "0 0 0 rg\n"; // RGB negro
614
615 // Definir área de recorte y dibujo con margen extra para mejor contraste
616 contenido << MARGEN_EXTRA << " " << MARGEN_EXTRA << " "
617 << (PAGE_WIDTH - 2 * MARGEN_EXTRA) << " "
618 << (PAGE_HEIGHT - 2 * MARGEN_EXTRA) << " re W n\n";
619
620 // Dibujar módulos del QR con mayor precisión
621 for (int y = 0; y < qr.getSize(); y++) {
622 for (int x = 0; x < qr.getSize(); x++) {
623 if (qr.getModule(x, y)) {
624 // Calcular posición con más precisión
625 double posX = MARGEN + x * MODULO_SIZE;
626 double posY = MARGEN + (qr.getSize() - 1 - y) * MODULO_SIZE;
627 // Usar un path más eficiente para cada módulo con tamaño exacto
628 contenido << std::fixed << std::setprecision(3)
629 << posX << " " << posY << " "
630 << MODULO_SIZE << " " << MODULO_SIZE << " re\n";
631 }
632 }
633 }
634 // Rellenar todos los módulos de una vez con máxima densidad
635 contenido << "f\n";
636 contenido << "Q\n"; // Restaurar estado gráfico
637
638 std::string contenidoStr = contenido.str();
639 objetos.push_back(pdf.tellp());
640 pdf << "4 0 obj\n"
641 << "<</Length " << contenidoStr.length() << ">>\nstream\n"
642 << contenidoStr
643 << "endstream\nendobj\n";
644
645 // xref
646 size_t xrefPos = pdf.tellp();
647 pdf << "xref\n0 " << (objetos.size() + 1) << "\n";
648 pdf << "0000000000 65535 f\r\n"; // Usar CRLF para mejor compatibilidad
649 for (size_t offset : objetos) {
650 pdf << std::setfill('0') << std::setw(10) << offset << " 00000 n\r\n";
651 }
652
653 // Añadir línea en blanco después del xref
654 pdf << "\n";
655
656 // trailer
657 pdf << "trailer\n"
658 << "<</Size " << (objetos.size() + 1) << " /Root 1 0 R>>\n"
659 << "startxref\n"
660 << xrefPos << "\n"
661 << "%%EOF\n";
662
663 pdf.close();
664 std::cout << "PDF generado exitosamente: " << nombreArchivo << std::endl;
665 }
